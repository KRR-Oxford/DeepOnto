[docs]    def load_dataset(self, data: List[Tuple[str, str, Any]]) -> Dataset:
        """For sequence classification, we have two sentences (sent1, sent2, cls_label) as input
        """
        data_df = pd.DataFrame(data, columns=["sent1", "sent2", "labels"])
        dataset = Dataset.from_pandas(data_df)
        dataset = dataset.map(
            lambda examples: self.tokenizer.tkz(
                examples["sent1"],
                examples["sent2"],
                max_length=self.args.max_length,
                truncation=True,
            ),
            batched=True,
            batch_size=1024,
            num_proc=10,
        )
        return dataset
